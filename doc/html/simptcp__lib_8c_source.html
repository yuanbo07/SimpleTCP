00146 int create_simptcp_socket()
00147 {
00148   int fd;
00149 
00150   /* get a free simptcp socket descriptor */
00151   for (fd=0;fd< MAX_OPEN_SOCK;fd++) {
00152     if ((simptcp_entity.simptcp_socket_descriptors[fd]) == NULL){ 
00153       /* this is a free descriptor */
00154       /* Allocating memory for the new simptcp_socket */
00155       simptcp_entity.simptcp_socket_descriptors[fd] = 
00156         (struct simptcp_socket *) malloc(sizeof(struct simptcp_socket));
00157       if (!simptcp_entity.simptcp_socket_descriptors[fd]) {
00158         return -ENOMEM;
00159       }
00160       /* initialize the simptcp socket control block with
00161          local port number set to 15000+fd */
00162       init_simptcp_socket(simptcp_entity.simptcp_socket_descriptors[fd],15000+fd);
00163       simptcp_entity.open_simptcp_sockets++;
00164       
00165       /* return the socket descriptor */
00166       return fd;
00167     }
00168   } /* for */
00169   /* The maximum number of open simptcp
00170      socket reached  */
00171   return -ENFILE;
